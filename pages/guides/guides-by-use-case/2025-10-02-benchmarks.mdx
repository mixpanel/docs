// Switcher lets you preview two visual styles without changing content
function StyleSwitcher({ style, setStyle }: { style: "minimal" | "feature"; setStyle: (s: "minimal" | "feature") => void }) {
  return (
    <div className="inline-flex items-center gap-1 rounded-full border border-slate-200 bg-white p-1 text-sm shadow-sm">
      {(["minimal", "feature"] as const).map((s) => (
        <button
          key={s}
          onClick={() => setStyle(s)}
          className={`rounded-full px-3 py-1 transition ${
            style === s ? "bg-slate-900 text-white" : "text-slate-700 hover:bg-slate-100"
          }`}
        >
          {s === "minimal" ? "Minimal cards" : "Feature tiles"}
        </button>
      ))}
    </div>
  );
}
